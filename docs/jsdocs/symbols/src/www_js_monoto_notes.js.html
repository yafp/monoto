104 /**
105  * @name initCKEditor
106  * @description initialize CKEditor
107  * @memberof notes
108  */
109 function initCKEditor()
110 {
111     console.debug("initCKEditor ::: Start");
112     console.log("initCKEditor ::: Initializing the CKEditor");
113 
114     // Defining the editor height
115     var monotoEditorHeight = 300; // setting a default value, in case there is non stored in localStorage
116     if(typeof(Storage)!=="undefined") // if localStorage is supported
117     {
118         monotoEditorHeight = window.localStorage.getItem("monotoEditorHeight");
119         console.log("initCKEditor ::: CKEditor height is set in localStorage to: "+monotoEditorHeight);
120     }
121     else
122     {
123         console.log("initCKEditor ::: CKEditor height is set to default value of: "+monotoEditorHeight);
124     }
125 
126     // START CKEDITOR
127     CKEDITOR.replace( "editor1",
128     {
129         // show all editor commands
130         //console.log(CKEDITOR.instances.editor1.commands);
131 
132         // key press handling
133         on:
134         {
135             instanceReady: function()
136             {
137                 CKEDITOR.instances.editor1.setReadOnly(true); // set RO mode
138                 CKEDITOR.config.toolbarCanCollapse = true; /* Enable collapse function for toolbar */
139                 return;
140             },
141 
142             focus: function(event)
143             {
144                 console.log("initCKEditor ::: CKEditor got focus");
145                 CKEDITOR.instances.editor1.execCommand( "toolbarCollapse", true ); // #241
146                 return;
147             },
148 
149             resize: function(event)
150             {
151                 // get new height of editor window
152                 editorHeight = event.editor.ui.space( "contents" ).getStyle( "height" );
153                 console.log("saveCKEditorHeightOnChange ::: CKEditor resized to: "+editorHeight);
154 
155                 //save to localstorage
156                 window.localStorage.setItem("monotoEditorHeight", editorHeight);
157             },
158 
159             key: function(e)
160             {
161                 setTimeout(function()
162                 {
163                     // #236 - KeyPress handling for ckeditor
164                     //console.log(e.data.keyCode);
165                     switch(e.data.keyCode)
166                     {
167                         // TAB -
168                         case 9:
169                             console.log("initCKEditor ::: Pressed TAB in CKEditor");
170                             // if CREATE button is visible -> give it focus
171                             if( $("#bt_createNewNote").is(":visible"))
172                             {
173                                 $("#bt_createNewNote").focus();
174                                 console.log("initCKEditor ::: Setting focus to create button");
175                                 break;
176                             }
177 
178                             // if SAVE button is visible -> give it focus
179                             if( $("#bt_saveNote").is("[disabled=disabled]"))
180                             {
181                                 $("#bt_saveNote").focus();
182                                 console.log("initCKEditor ::: Setting focus to save button");
183                                 break;
184                             }
185 
186                             // any other case
187                             $("#noteTitle").focus();
188                             console.log("initCKEditor ::: Setting focus to note title");
189                             break;
190 
191                         case 27: // ESC: - reset
192                             console.log("initCKEditor ::: Pressed ESC in CKEditor");
193                             //resetNotesUI();
194                             // set focus back to searchField
195                             $("#searchField").focus(); // as arrow up/down needs focus to work
196                             break;
197 
198                         // F2 - trigger maximize editor to fullscreen
199                         case 113:
200                             console.log("initCKEditor ::: Pressed F2 in CKEditor");
201                             CKEDITOR.instances.editor1.execCommand( "maximize" );
202                             break;
203 
204                         //default:
205                             //console.log("initCKEditor ::: Default case");
206                     }
207                 },1);
208             } // end: key
209         }, // end: on
210 
211         // other things
212         //
213         enterMode: CKEDITOR.ENTER_BR,
214         height: monotoEditorHeight,
215         toolbarCanCollapse: true, // enable collapse option
216         toolbarStartupExpanded : false,  // define collapsed as default
217         //removePlugins: 'elementspath',
218         toolbar:
219         [
220             { name: "tools",       items : [ "Maximize" ] },
221             { name: "basicstyles", items : [ "Bold","Italic","Strike","RemoveFormat" ] },
222             { name: "paragraph",   items : [ "NumberedList","BulletedList","-","Outdent","Indent","Blockquote" ] },
223             { name: "insert",      items : [ "Link","Image","Table","HorizontalRule","SpecialChar" ] },
224             { name: "styles",      items : [ "Styles","Format" ] },
225             { name: "document",    items : [ "Source" ] }
226         ]
227     });
228 
229     printCKEditorStatus();
230     console.debug("initCKEditor ::: Stop");
231 }

690         });
691     }
